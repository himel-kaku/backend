** database:


 TABLE department (
    dept_id INT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    dept_code VARCHAR(20) UNIQUE NOT NULL,
    building_name VARCHAR(100)
);

 TABLE student (
    student_id INT PRIMARY KEY,
    stu_name VARCHAR(100) NOT NULL,
    level INT,
    term INT,
    role VARCHAR(50),
    department_id INT REFERENCES department(dept_id), -- Fixed reference
    password VARCHAR(25)
);

 TABLE teacher (
    teacher_id INT PRIMARY KEY,
    teacher_name VARCHAR(100) NOT NULL,
    department_id INT REFERENCES department(dept_id), -- Fixed reference
    password VARCHAR(25)
);

 TABLE course (
    id SERIAL PRIMARY KEY,
    course_code VARCHAR(20) UNIQUE NOT NULL,
    course_title VARCHAR(100) NOT NULL,
    course_type VARCHAR(50),
    department_id INT REFERENCES department(dept_id) -- Fixed reference
);

 TABLE student_course (
    student_id INT REFERENCES student(student_id), -- Fixed reference
    course_id INT REFERENCES course(id),
    section VARCHAR(10),
    PRIMARY KEY (student_id, course_id)
);


-- TABLE course and teacher
 TABLE course_teacher (
    course_id INT REFERENCES course(id),
    teacher_id INT REFERENCES teacher(teacher_id), -- Fixed reference
    PRIMARY KEY (course_id, teacher_id)
);

-- Table: class_schedule for showing a class routine; each schedule is a block of routine. like a date block of calender
 TABLE class_schedule (
    id SERIAL PRIMARY KEY,
    course_id INT REFERENCES course(id) ON DELETE CASCADE,
    section VARCHAR(5),
    date DATE,
    start_time TIME,
    end_time TIME,
    building_name VARCHAR(50),
    room VARCHAR(25),
    is_regular SMALLINT DEFAULT 1
);

-- Table: exam_schedule; I will keep a rows of dates; and where each date will contains some blocks of exam schedule
 TABLE exam_schedule (
    id SERIAL PRIMARY KEY,
    course_id INT REFERENCES course(id) ON DELETE CASCADE,
    section VARCHAR(5),
    date DATE,
    start_time TIME,
    end_time TIME,
    building_name VARCHAR(50),
    room VARCHAR(25),
    exam_description TEXT
);

-- Table: class_plots; suppose, I my website started at 25january, then I will keep plots from 25jan to today date for each course
 TABLE class_plots (
    id SERIAL PRIMARY KEY,
    course_id INT REFERENCES course(id) ON DELETE CASCADE,
    section VARCHAR(5),
    date DATE
);

-- Table: exam_plots; for storing past and present exam files information
 TABLE exam_plots (
    id SERIAL PRIMARY KEY,
    course_id INT REFERENCES course(id) ON DELETE CASCADE,
    section VARCHAR(5),
    date DATE
);


-- -- Table: uploaded_files; track which files are uploaded on a specific plot [class_plot or exam_plot]
 TABLE uploaded_files (
    id SERIAL PRIMARY KEY,
    file_name VARCHAR(200) NOT NULL,
    file_path VARCHAR(500) NOT NULL,
    uploaded_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    plot_id INT NOT NULL,       -- refers to class_plots.id or exam_plots.id
    is_plot_class SMALLINT NOT NULL CHECK (is_plot_class IN (0, 1)) -- 1 = class, 0 = exam
);
